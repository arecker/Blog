def generate(site)
  @site = Site.new(site)
  if @site.production?
    info "skipping #{name} generator"
  else
    info "running #{name} generator"
    data = crunch
    File.open(data_file_target, 'w') { |f| f.write(JSON.pretty_generate(data)) } unless data.nil?
  end
end
