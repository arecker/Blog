def write
  g = ::Gruff::Pie.new('800x600')
  g.theme = Gruff::Themes::PASTEL
  g.hide_legend = false
  g.legend_at_bottom = true
  g.minimum_value = 0
  results.each { |w, n| g.data w, n }
  g.write(graphs_join('swears.png'))
end
